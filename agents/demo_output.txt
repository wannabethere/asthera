WITH
step_1_output AS (
SELECT division, completed_date FROM csod_training_records WHERE is_completed = true AND completed_date >= DATE_TRUNC('month', CURRENT_DATE) - INTERVAL '6 months' AND completed_date < DATE_TRUNC('month', CURRENT_DATE)
),
step_2_output AS (
WITH bucket_counts AS (SELECT division, completed_date::DATE AS date_bucket, COUNT(*) AS cnt FROM step_1_output GROUP BY division, completed_date::DATE) SELECT division, json_agg(json_build_object('time', date_bucket, 'metric', cnt) ORDER BY date_bucket)::JSONB AS metric_series FROM bucket_counts GROUP BY division
),
step_3_output AS (
SELECT s.division, fn.* FROM step_2_output AS s CROSS JOIN LATERAL calculate_sma(s.metric_series, 7) AS fn
),
step_4_output AS (
SELECT s.division, fn.* FROM step_2_output AS s CROSS JOIN LATERAL detect_anomalies(s.metric_series, 'zscore', 2.0) AS fn
)
SELECT
    p.division,
    p.time_period,
    p.sma_value,
    p.upper_band,
    p.lower_band,
    j0.value,
    j0.anomaly_type,
    j0.anomaly_score
FROM step_3_output p
JOIN step_4_output j0 ON p.division = j0.division AND p.time_period = j0.time_period
WHERE j0.anomaly_type != 'normal'
ORDER BY p.division, p.time_period
